import matplotlib.pyplot as plt
import matplotlib.animation as animation
import numpy as np
from IPython.display import Image # Tambahkan ini untuk nampilin GIF di notebook

# Inisialisasi Figure
fig, ax = plt.subplots(figsize=(6, 8))

# Titik koordinat persegi ABCD
x_obj = np.array([2, 2, 3, 3, 2])
y_obj_base = np.array([3, 4, 4, 3, 3])

# Pengaturan ruang grafik
ax.set_xlim(-1, 6)
ax.set_ylim(-6, 6)
ax.set_xticks(np.arange(-1, 7, 1))
ax.set_yticks(np.arange(-6, 7, 1))
ax.grid(True, which='both', linestyle='--', linewidth=0.5, color='gray', alpha=0.7)

# Garis sumbu X (Cermin) dan Y
ax.axhline(y=0, color='green', linewidth=2, linestyle='-.', label='Cermin (Sumbu X)')
ax.axvline(x=0, color='black', linewidth=1.5)

# Inisialisasi garis
line_obj, = ax.plot([], [], 'b-', linewidth=2, label='Objek (ABCD)')
line_ref, = ax.plot([], [], 'r--', linewidth=2, label='Refleksi (IJKL)')

# Label teks
text_obj = ax.text(0, 0, 'ABCD', color='blue', ha='center', va='bottom', fontweight='bold')
text_ref = ax.text(0, 0, 'IJKL', color='red', ha='center', va='top', fontweight='bold')

ax.legend(loc='upper right')
title = ax.set_title("Animasi Refleksi")

def init():
    line_obj.set_data([], [])
    line_ref.set_data([], [])
    text_obj.set_position((-10, 0))
    text_ref.set_position((-10, 0))
    return line_obj, line_ref, text_obj, text_ref

def animate(i):
    shift_y = np.cos(i * np.pi / 50) - 1       
    y_obj = y_obj_base + shift_y
    y_ref = -y_obj 

    line_obj.set_data(x_obj, y_obj)
    line_ref.set_data(x_obj, y_ref)
    
    text_obj.set_position((2.5, np.max(y_obj) + 0.2)) 
    text_ref.set_position((2.5, np.min(y_ref) - 0.2)) 

    title.set_text(f"Frame: {i} | Refleksi terhadap Sumbu X")
    return line_obj, line_ref, text_obj, text_ref

# Membuat objek animasi
ani = animation.FuncAnimation(
    fig, animate, init_func=init, 
    frames=100, interval=40, blit=True
)

# --- BAGIAN PERUBAHAN UTAMA ---

# 1. Simpan sebagai GIF (Pastikan writer='pillow' sudah terpasang)
ani.save('animasi_refleksi.gif', writer='pillow', fps=25)

# 2. Tutup figure agar tidak muncul gambar statis tambahan
plt.close(fig) 

# 3. Tampilkan GIF-nya di Jupyter Lab (opsional, biar kamu bisa cek)
Image(open('animasi_refleksi.gif','rb').read())
